// ✅ Cómo debes pensarlo en Angular
// "Necesito un servicio con providedIn correcto"

@Injectable({
  providedIn: 'root'  // Singleton en toda la app — instancia única
})
export class AuthService {
  private currentUser = signal<User | null>(null);
  
  // signal() aquí es el equivalente a useState() pero compartido
  // La diferencia clave: este signal vive en el servicio, no en el componente
  readonly user = this.currentUser.asReadonly();
  
  login(credentials: LoginDto): Observable<User> {
    return this.http.post<User>('/api/auth/login', credentials).pipe(
      tap(user => this.currentUser.set(user))
    );
  }
}
```

**Por bloques:**

`@Injectable({ providedIn: 'root' })` — Esto le dice al injector raíz de Angular "registra este servicio como singleton".
 No necesitas importarlo en ningún módulo. Angular lo incluirá en el bundle solo si alguien lo consume (tree-shakeable).
```
